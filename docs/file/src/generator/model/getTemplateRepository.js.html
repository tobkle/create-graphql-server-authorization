    src/generator/model/getTemplateRepository.js

    // @flow

import fs from 'fs';
import path from 'path';
import _ from 'lodash';
import { addTemplateToRepository } from './addTemplateToRepository';

import { TEMPLATE_EXTENSION, ENCODING, DELIMITER } from '../../constants';

/**
 * reads all available templates of a template directory
 * @public
 * @param {string} templates_dir - templates directory name
 * @return {Object} template_repository - repository with all templates
 *
 * @example
 *  repository = {
 *    "header": <template source> 
 *    "class": <template source>
 *    "methods": {
 *      "constructor": <template source>,
 *      "authRole": <template source>,
 *      "findOneById": <template source>,
 *      "find": <template source>,
 *      ... : ...
 *      "insert": <template source>,
 *      "updateById": <template source>,
 *      "removeById": <template source>,
 *    }
 *  }
 */

export function getTemplateRepository(templates_dir: string): mixed {
  let repository = {};
  const templates_directory = path.join(templates_dir);
  // read templates directory and get all template names
  fs
    .readdirSync(templates_directory)
    .filter(file => path.extname(file) === TEMPLATE_EXTENSION)
    .sort()
    .forEach(file => {
      // get template path
      const templateFilename = path.join(templates_dir, file);
      // read template source
      const templateSource = fs.readFileSync(templateFilename, ENCODING);

      // split template name for repository creation
      const templateName = path
        .basename(file, TEMPLATE_EXTENSION)
        .split(DELIMITER);

      // remove sequence number
      templateName.shift();

      // add template to repository
      repository = _.merge(
        repository,
        addTemplateToRepository(templateName, templateSource)
      );
    });

  return repository;
}
